        // Exponential Backoff Implementation (Up to 5 retries: 1s, 2s, 4s, 8s, 16s)

        async function makeApiCallWithBackoff(url, payload, retries = 5, delay = 1000) {

            const retryInd = document.getElementById('retryIndicator');

            for (let i = 0; i < retries; i++) {

                try {

                    const response = await fetch(url, {

                        method: 'POST',

                        headers: {

                            'Content-Type': 'application/json'

                        },

                        body: JSON.stringify(payload)

                    });

                    if (!response.ok) {

                        const errorData = await response.json();

                        throw new Error(errorData.error?.message || `HTTP ${response.status}`);

                    }

                    const data = await response.json();

                    const text = data.candidates?.[0]?.content?.parts?.[0]?.text;

                    if (!text) {

                        throw new Error("Format respons API kosong");

                    }

                    return text;

                } catch (error) {

                    if (i === retries - 1) throw error; // If last retry failed, throw error

                   

                    // Show retry state in UI

                    retryInd.classList.remove('hidden');

                    await new Promise(res => setTimeout(res, delay));

                    delay *= 2; // Double the wait time

                }

            }

            retryInd.classList.add('hidden');

        }

        // Import Database from JSON

        function importData(input) {

            const file = input.files[0];

            if (!file) return;

            const reader = new FileReader();

            reader.onload = function(e) {

                try {

                    const imported = JSON.parse(e.target.result);

                    if (Array.isArray(imported)) {

                        // Merge on no_telepon primary key

                        imported.forEach(newItem => {

                            if (!newItem.no_telepon) return;

                            const idx = database.findIndex(x => x.no_telepon === newItem.no_telepon);

                            if (idx !== -1) {

                                database[idx] = newItem; // overwrite

                            } else {

                                database.push(newItem);

                            }

                        });

                        saveToLocalStorage();

                        renderTable();

                        showToast(`Impor berhasil! Total database saat ini: ${database.length} entri.`, "success");

                    } else {

                        throw new Error("Format file JSON harus berupa Array");

                    }

                } catch (err) {

                    showToast(`Gagal mengimpor file: ${err.message}`, "error");

                }

                input.value = ''; // Reset input file

            };

            reader.readAsText(file);

        }
